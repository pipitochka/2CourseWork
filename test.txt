#include <stdio.h>

void calculator() {
    double num1, num2, result;
    char operator;

    printf("Введите выражение (например: 5 + 3): ");
    if (scanf("%lf %c %lf", &num1, &operator, &num2) != 3) {
        printf("Некорректный ввод.\n");
        return;
    }

    switch (operator) {
        case '+':
            result = num1 + num2;
            printf("Результат: %.2f\n", result);
            break;
        case '-':
            result = num1 - num2;
            printf("Результат: %.2f\n", result);
            break;
        case '*':
            result = num1 * num2;
            printf("Результат: %.2f\n", result);
            break;
        case '/':
            if (num2 == 0) {
                printf("Ошибка: деление на ноль.\n");
            } else {
                result = num1 / num2;
                printf("Результат: %.2f\n", result);
            }
            break;
        default:
            printf("Некорректный оператор: %c\n", operator);
    }
}

int main() {
    printf("Добро пожаловать в калькулятор!\n");
    calculator();
    return 0;
}
